/**
 * DATABASE CONNECTOR: Dynamically binds to the workbook safely across all execution contexts
 */
function getSpreadsheetTemplatesConfig() {
  try {
    let ss = SpreadsheetApp.getActiveSpreadsheet();
    if (!ss) {
      ss = SpreadsheetApp.openById("1nkHEdI7M96BlHnA91_Xpy_ReLbmc_Yp6gjaORGIELsA");
    }
    
    const sheet = ss.getSheetByName("Email Templates");
    if (!sheet) throw new Error("Could not find the 'Email Templates' tab inside this workbook.");
    
    const rows = sheet.getDataRange().getValues();
    const headers = rows[0].map(h => String(h).trim());
    
    const keyIdx = headers.indexOf("Template_Key");
    const labelIdx = headers.indexOf("Dropdown_Label");
    const subjectIdx = headers.indexOf("Default_Subject");
    const titleIdx = headers.indexOf("Default_Title");
    const rawNotesIdx = headers.indexOf("Dashboard_Notes_Raw");
    const htmlBodyIdx = headers.indexOf("Email_Body_HTML");
    const downtimeFlagIdx = headers.indexOf("Requires_Downtime"); 
    
    let templatesArray = [];
    for (let i = 1; i < rows.length; i++) {
      let key = String(rows[i][keyIdx]).trim();
      if (!key) continue;
      
      templatesArray.push({
        key: key,
        label: String(rows[i][labelIdx]).trim(),
        subject: String(rows[i][subjectIdx]).trim(),
        title: String(rows[i][titleIdx]).trim(),
        rawNotes: String(rows[i][rawNotesIdx]).trim().replace(/\\n/g, '\n'),
        htmlBody: String(rows[i][htmlBodyIdx]).trim(),
        requiresDowntime: downtimeFlagIdx !== -1 ? (String(rows[i][downtimeFlagIdx]).trim().toUpperCase() === "TRUE") : false
      });
    }
    return { status: 'success', templates: templatesArray };
  } catch (err) {
    return { status: 'error', message: err.toString() };
  }
}

/**
 * DYNAMIC POD DATA INGESTION ENGINE
 */
function fetchPodDataFromSheet() {
  try {
    let rows;
    try {
      let ss = SpreadsheetApp.getActiveSpreadsheet();
      if (!ss) {
        ss = SpreadsheetApp.openById("1nkHEdI7M96BlHnA91_Xpy_ReLbmc_Yp6gjaORGIELsA");
      }
      
      let sheet = ss.getSheets()[0];
      const targetSheets = ss.getSheets().filter(s => s.getSheetId() == 1274836295);
      if (targetSheets.length > 0) sheet = targetSheets[0];
      
      const lastRow = sheet.getLastRow();
      const lastCol = sheet.getLastColumn();
      rows = sheet.getRange(1, 1, lastRow, lastCol).getValues();
    } catch (sheetErr) {
      console.error("Sheet baseline disconnect handled: " + sheetErr.toString());
      return { 
        status: 'success', 
        individualPods: ["VV3-3057", "VV3-3099"], 
        podMetadata: {"VV3-3057": {region: "APAC", release: "General", product: "Vault", trustId: "", upgradeMode: "Classic"}},
        uniqueRegions: ["US", "EU", "APAC"],
        uniqueReleases: ["General"],
        uniqueProducts: ["Safety", "LIMS", "RegOne", "Vault"]
      };
    }

    const headers = rows[0].map(h => String(h).trim().toUpperCase());
    
    const podColIndex = headers.indexOf("POD");          
    const regionColIndex = headers.indexOf("REGION");    
    const releaseColIndex = headers.indexOf("RELEASE");  
    const productColIndex = headers.indexOf("PRODUCT");
    const podGroupColIndex = headers.indexOf("VAULT POD GROUP"); 
    const trustIdColIndex = headers.indexOf("TRUST ID"); 
    const upgradeModeColIndex = headers.indexOf("UPGRADE MODE");
    
    let podValues = [];
    let podMetadata = {};
    let regionsSet = new Set();
    let releasesSet = new Set();
    let productsSet = new Set();

    for (let i = 1; i < rows.length; i++) {
      let currentPod = (podColIndex !== -1) ? String(rows[i][podColIndex]).trim() : "";
      let currentRegion = (regionColIndex !== -1) ? String(rows[i][regionColIndex]).trim() : "";
      let currentRelease = (releaseColIndex !== -1) ? String(rows[i][releaseColIndex]).trim() : "";
      let currentProduct = (productColIndex !== -1) ? String(rows[i][productColIndex]).trim() : "";
      let currentCategory = (podGroupColIndex !== -1) ? String(rows[i][podGroupColIndex]).trim() : "";
      let currentTrustId = (trustIdColIndex !== -1) ? String(rows[i][trustIdColIndex]).trim() : "";
      let currentUpgradeMode = (upgradeModeColIndex !== -1) ? String(rows[i][upgradeModeColIndex]).trim() : "";

      if (currentPod === "" || /^\d+$/.test(currentPod) || currentCategory === "") {
        continue;
      }

      podValues.push(currentPod);
      
      if (!currentRegion) currentRegion = "Unknown";
      if (!currentRelease) currentRelease = "Unknown";
      if (!currentProduct) currentProduct = "Unknown";
      if (!currentUpgradeMode) currentUpgradeMode = "Unknown";

      if (currentRegion !== "NA") regionsSet.add(currentRegion);
      releasesSet.add(currentRelease);
      
      if (currentProduct !== "Unknown" && currentProduct !== "Vault Basics" && currentProduct !== "") {
        productsSet.add(currentProduct);
      }

      podMetadata[currentPod] = {
        region: currentRegion,
        release: currentRelease,
        product: currentProduct,
        trustId: currentTrustId,
        upgradeMode: currentUpgradeMode
      };
    }
    
    let finalUniqueProducts = [...productsSet].sort();
    if (finalUniqueProducts.length === 0) {
      finalUniqueProducts = ["Safety", "LIMS", "RegOne", "Vault"];
    }
    
    return {
      status: 'success',
      individualPods: [...new Set(podValues)].sort((a,b) => String(a).localeCompare(String(b), undefined, {numeric: true})),
      podMetadata: podMetadata,
      uniqueRegions: [...regionsSet].sort(),
      uniqueReleases: [...releasesSet].sort(),
      uniqueProducts: finalUniqueProducts,
      footerHtml: getUnifiedFooterText()
    };
  } catch (err) {
    return { status: 'error', message: err.toString() };
  }
}

function getSenderIdByEmail(apiKey, targetEmail) {
  const options = {
    "method": "get",
    "headers": { "Authorization": "Bearer " + apiKey },
    "muteHttpExceptions": true
  };
  
  const response = UrlFetchApp.fetch("https://api.sendgrid.com/v3/verified_senders", options);
  if (response.getResponseCode() !== 200) {
    throw new Error("Unable to read Sender Identity records from SendGrid: " + response.getContentText());
  }
  
  const data = JSON.parse(response.getContentText());
  const sendersList = data.results || data || [];
  const match = sendersList.find(s => s.from_email && s.from_email.toLowerCase() === targetEmail.toLowerCase());
  if (!match) throw new Error("Could not find a SendGrid verified identity matching: " + targetEmail);
  return match.id;
}

/**
 * UNIFIED HTML GENERATOR LAYOUT
 */
function buildUnifiedHtmlBody(formData, isStatusCast) {
  const isDowntimeRequired = formData.requiresDowntime === true || formData.requiresDowntime === "true";
  
  const defaultSubject = isDowntimeRequired
    ? "Veeva Release Notification" 
    : "Veeva Vault Maintenance Release Notification";

  const defaultVersion = "26R1.0.22";
  const defaultDowntime = "90 Minutes";
  const defaultFriendlyDate = "Wednesday, April 29 at 10:00 am PDT";
  const defaultUrl = "https://vaulthelp.veeva.com";
  const defaultIso = "20260429T10";
  
  const cleanTitleHeader = (formData.emailTitle && formData.emailTitle.trim() !== "") ? formData.emailTitle.trim() : defaultSubject;
  const rawVersion = (formData.version && formData.version.trim() !== "") ? formData.version.trim() : defaultVersion;
  const finalDowntimeString = (formData.estimatedDowntime && formData.estimatedDowntime.trim() !== "") ? formData.estimatedDowntime.trim() : defaultDowntime;
  const cleanFriendlyDate = (formData.friendlyDate && formData.friendlyDate.trim() !== "") ? formData.friendlyDate.trim() : defaultFriendlyDate;
  const cleanUrl = (formData.notesUrl && formData.notesUrl.trim() !== "") ? formData.notesUrl.trim() : defaultUrl;
  const cleanIso = (formData.isoString && formData.isoString.trim() !== "") ? formData.isoString.trim() : defaultIso;
  const cleanEnvTheme = formData.envTheme || "PROD";

  let topBannerImageUrl = cleanEnvTheme === "SBX" 
    ? "http://cdn.mcauto-images-production.sendgrid.net/f3de798f1d5c4acf/4b9a6798-b8ad-475d-940b-67aaeb29039f/1578x93.png"
    : "http://cdn.mcauto-images-production.sendgrid.net/f3de798f1d5c4acf/39e112bf-4c92-4bcb-9fca-cf442dfbff2d/776x50.png";

  let bottomBannerImageUrl = cleanEnvTheme === "SBX"
    ? "http://cdn.mcauto-images-production.sendgrid.net/f3de798f1d5c4acf/f8bb1bde-21f8-4910-a6b4-ae373a014261/1578x50.png"
    : "http://cdn.mcauto-images-production.sendgrid.net/f3de798f1d5c4acf/66c81663-cafb-4ec2-851d-b6c10e25dfba/779x25.png";

  const topBannerMarkup = `<img src="${topBannerImageUrl}" alt="Top Banner" style="width: 100%; max-width: 650px; height: auto; display: block;" />`;
  const bottomBannerMarkup = `<img src="${bottomBannerImageUrl}" alt="Bottom Banner" style="width: 100%; max-width: 650px; height: auto; display: block;" />`;

  let description = "The Veeva Vault is scheduled to be upgraded to the latest maintenance release. There is no expected downtime, however, a short service interruption may be experienced.";
  let dynamicDowntimeRowMarkup = "";
  if (isDowntimeRequired) {
    description = "Veeva Platform will be upgraded to the latest maintenance release and the system will be <strong>unavailable for use during this time.</strong>";
    dynamicDowntimeRowMarkup = `<tr><td style="padding: 4px 0;"><strong>Estimated Vault Downtime:</strong> ` + finalDowntimeString + `</td></tr>`;
  }

  let regionHeaderPrefix = "PODs";
  let formattedPodsListString = "";

  if (formData.affectedPods) {
    let upperPodsText = formData.affectedPods.replace(/<\/?[9>]+(>|$)/g, "").toUpperCase();
    let selectedRegionsSet = new Set();
    
    if (upperPodsText.includes("US ")) selectedRegionsSet.add("US");
    if (upperPodsText.includes("EU ")) selectedRegionsSet.add("EU");
    if (upperPodsText.includes("APAC ")) selectedRegionsSet.add("APAC");

    const strictRegionOrder = ["US", "EU", "APAC"];
    let orderedRegions = strictRegionOrder.filter(r => selectedRegionsSet.has(r));
    if (orderedRegions.length > 0) {
      regionHeaderPrefix = orderedRegions.join('/') + " PODs";
    }

    let lines = formData.affectedPods.split(/[\n\r]+/).map(l => l.trim()).filter(Boolean);
    formattedPodsListString = lines.join('<br/>');
  }

  let shortVersion = rawVersion.substring(0, 4);
  // 🚀 OPTIMIZATION: Pulls from spreadsheet Email_Body_HTML payload first, falls back to raw notes text
  let rawNotesText = (formData.htmlBody && formData.htmlBody.trim() !== "") ? formData.htmlBody : (formData.customNotesText || "");
  
  let tempHtml = String(rawNotesText).replace(/\n/g, '<br/>');
  
  let processedNotesHtml = tempHtml
    .replace(/\[\[LOCATION_TAG\]\]/g, `<a href="https://rn.veevavault.help/en/gr/vault-pod-details/" clicktracking="off" target="_blank" style="color: #007bff; text-decoration: none;">here</a>`)
    // 🚀 FIX: Evaluates brackets smoothly for templates that use it, but leaves direct sheet links pristine without injecting structural numbers
    .replace(/\[\[VERSION\]\] Release Notes/g, `<a href="${cleanUrl}" clicktracking="off" target="_blank" style="color: #007bff; text-decoration: none; font-weight: normal;">${shortVersion} Release Notes</a>`)
    .replace(/\[\[VERSION\]\]/g, shortVersion)
    .replace(/\$\{cleanUrl\}/g, cleanUrl)
    .replace(/(Vault's POD location\.)(?:<br\s*\/?>\s*)+/i, '$1<hr style="border: 0; border-top: 2px solid #b1b1b1; margin-top: 8px; margin-bottom: 8px;"/>')
    .replace(/Veeva Trust Site/g, "<a href='https://trust.veeva.com' clicktracking='off' target='_blank' style='color: #007bff; text-decoration: none;'>Veeva Trust Site</a>")
    .replace(/Veeva Product Support/g, "<a href='https://support.veeva.com/hc/en-us' clicktracking='off' target='_blank' style='color: #007bff; text-decoration: none;'>Veeva Product Support</a>");

  const worldClockUrl = "https://www.timeanddate.com/worldclock/fixedtime.html?p1=5402&amp;iso=" + cleanIso + "&amp;sort=1";

  let timeStringsArray = [];
  if (cleanIso && cleanIso.length >= 11) {
    try {
      let yr = cleanIso.substring(0,4);
      let mo = cleanIso.substring(4,6);
      let dy = cleanIso.substring(6,8);
      let hr = cleanIso.substring(9,11);
      let baseDate = new Date(`${yr}-${mo}-${dy}T${hr}:00:00`);
      
      if (!isNaN(baseDate.getTime())) {
        let upperPodsText = formData.affectedPods ? formData.affectedPods.toUpperCase() : "";
        let hasUS = upperPodsText.includes("US ");
        let hasEU = upperPodsText.includes("EU ");
        let hasAPAC = upperPodsText.includes("APAC ");
        
        if (!hasUS && !hasEU && !hasAPAC) { hasUS = true; hasEU = true; hasAPAC = true; }
        
        if (hasUS) {
          timeStringsArray.push(cleanFriendlyDate);
        }
        if (hasEU) {
          let dateCEST = new Date(baseDate.getTime() + (9 * 60 * 60 * 1000));
          let optionsCEST = { weekday: 'long', month: 'long', day: 'numeric', hour: 'numeric', minute: '2-digit' };
          let formattedCEST = dateCEST.toLocaleString('en-US', optionsCEST).replace(',', '').replace(' at ', ', ') + " CEST";
          timeStringsArray.push(formattedCEST);
        }
        if (hasAPAC) {
          let dateJST = new Date(baseDate.getTime() + (16 * 60 * 60 * 1000)); 
          let optionsJST = { weekday: 'long', month: 'long', day: 'numeric', hour: 'numeric', minute: '2-digit' };
          let formattedJST = dateJST.toLocaleString('en-US', optionsJST).replace(',', '').replace(' at ', ', ') + " JST";
          timeStringsArray.push(formattedJST);
        }
      }
    } catch(e) {
      console.error("Time token compiler error: " + e.toString());
    }
  }

  let finalWhenString = timeStringsArray.length > 0 ? timeStringsArray.join(' / ') : cleanFriendlyDate;

  return `
    <div style="font-family: Arial, Helvetica, sans-serif; max-width: 650px; margin: auto; color: #333333; background-color: #ffffff; padding: 10px;">
      ${topBannerMarkup}
      <div style="padding: 20px 20px 20px 20px;">
        <h2 style="text-align: center; color: #000000; font-size: 22px; font-weight: bold; margin-top: 0; margin-bottom: 20px; font-family: Arial, sans-serif;">${cleanTitleHeader}</h2>
        <p style="line-height: 1.5; font-size: 14px; color: #333333; margin-bottom: 25px;">${description}</p>
        <table style="width: 100%; border-collapse: collapse; font-size: 14px; font-family: Arial, sans-serif;">
          <tr><td style="padding: 4px 0;"><strong>What:</strong> Veeva Vault ${rawVersion}</td></tr>
          <tr><td style="padding: 4px 0;"><strong>When:</strong> <a href="${worldClockUrl}" clicktracking="off" target="_blank" style="color: #007bff; text-decoration: underline;">${finalWhenString}</a></td></tr>
          ${dynamicDowntimeRowMarkup}
          <tr><td style="padding: 4px 0;"><strong>Affected PODs:</strong> <span>${regionHeaderPrefix}</span></td></tr>
        </table>
        <div style="margin-top: 12px; font-weight: normal; font-size: 14px; line-height: 1.6; color: #333333; font-family: Arial, sans-serif; white-space: normal;">
          ${formattedPodsListString}
          ${formattedPodsListString ? '<br/><br/>' : ''}
          ${processedNotesHtml}
        </div>
      </div>
      ${bottomBannerMarkup}
      <p style="font-size: 11px; color: #555555; line-height: 1.5; padding: 15px; margin: 0; background-color: #ffffff; font-family: Arial, sans-serif;">
        ${getUnifiedFooterText()}
      </p>
    </div>
  `;
}

function processPushRequest(formData) {
  try {
    if (!formData) formData = {};
    const scriptProperties = PropertiesService.getScriptProperties();
    const SENDGRID_API_KEY = scriptProperties.getProperty('SendGridAPIKeyID');
    if (!SENDGRID_API_KEY) throw new Error("SendGrid API Key parameter context not discovered.");

    const rawVersion = formData.version || "26R1.0.22";
    const cleanFromSender = formData.fromSender || "no-reply@vault.veeva.com"; 

    const targetSenderId = getSenderIdByEmail(SENDGRID_API_KEY, cleanFromSender);
    
    const compiledTitleWithVersion = (formData.emailTitle || "").trim() + " - " + rawVersion.trim();
    const compiledSubjectWithVersion = (formData.subjectLine || "").trim() + " - " + rawVersion.trim();
    const finalHtmlBody = buildUnifiedHtmlBody(formData, false);

    const payload = {
      "name": compiledTitleWithVersion, 
      "email_config": { 
        "subject": compiledSubjectWithVersion, 
        "html_content": finalHtmlBody, 
        "generate_plain_content": true,
        "custom_unsubscribe_url": "https://platform.veevavault.help/en/lr/7239/#notification-email",
        "sender_id": targetSenderId,
        "tracking_settings": {
          "open_tracking": { "enable": false },
          "click_tracking": { "enable": false }
        }
      }
    };

    const options = {
      "method": "post",
      "headers": { "Authorization": "Bearer " + SENDGRID_API_KEY, "Content-Type": "application/json" },
      "payload": JSON.stringify(payload),
      "muteHttpExceptions": true
    };

    const response = UrlFetchApp.fetch("https://api.sendgrid.com/v3/marketing/singlesends", options);
    if (response.getResponseCode() >= 400) throw new Error("SendGrid Refused Payload: " + response.getContentText());
    return { status: 'success' };
  } catch (err) {
    throw new Error(err.message);
  }
}

function processTrustPushRequest(formData) {
  try {
    if (!formData) formData = {};
    var scriptProperties = PropertiesService.getScriptProperties();
    
    var rawApiKey = scriptProperties.getProperty('StatusCastProdTrustAPIKey');
    var fullDomain = scriptProperties.getProperty('StatusCastPageID'); 
    
    if (!rawApiKey || !fullDomain) {
      throw new Error("Required configurations missing from Script Properties.");
    }
    
    var apiKey = String(rawApiKey).trim();
    var authorizationValue = apiKey.toLowerCase().startsWith("bearer ") ? apiKey : "Bearer " + apiKey;
    
    var componentsUrl = "https://" + String(fullDomain).trim() + "/api/v4/components";
    var componentsResponse = UrlFetchApp.fetch(componentsUrl, {
      "method": "get",
      "headers": { "Authorization": authorizationValue, "accept": "application/json" },
      "muteHttpExceptions": true
    });

    if (componentsResponse.getResponseCode() !== 200) {
      throw new Error("Failed to authenticate or download active components map from StatusCast API.");
    }

    var activeComponentsData = JSON.parse(componentsResponse.getContentText());
    var validStatusCastIds = new Set();
    
    if (Array.isArray(activeComponentsData)) {
      activeComponentsData.forEach(function(comp) {
        if (comp.id) validStatusCastIds.add(Number(comp.id));
      });
    }

    var incidentUrl = "https://" + String(fullDomain).trim() + "/api/v4/incident";
    var now = new Date();
    var currentIsoDate = now.toISOString();
    var timeToken = now.toTimeString().split(' ')[0];
    
    var currentUserEmail = "your-statuscast-admin-email@company.com"; 
    var userPrefix = "System"; 
    
    var baseTitle = formData.emailTitle || "";
    var baseSubject = formData.subjectLine || "";
    var rawVersion = formData.version || "26R1.0.22";
    var statusType = formData.statusCastType || "Maintenance";
    
    var shortTag = "UNV";
    if (statusType === "Performance Degraded" || statusType === "Performance Degradation") shortTag = "DEG";
    if (statusType === "Maintenance") shortTag = "MNT";
    if (statusType === "Informational") shortTag = "INF";
    
    var uniqueTitle = shortTag + ": " + baseTitle.trim() + " - " + rawVersion + " (" + timeToken + ") [via " + userPrefix + "]";
    var customSubjectLine = baseSubject.trim() + " - " + rawVersion;

    var sheetData = fetchPodDataFromSheet();
    var realComponentIds = [];
    var droppedComponentLog = [];
    var affectedPodsText = formData.affectedPods || "";
    
    if (sheetData && sheetData.status === 'success' && sheetData.podMetadata) {
      var cleanTextAreaContent = affectedPodsText.replace(/<\/?[9>]+(>|$)/g, " ").replace(/[\r\n]+/g, " ").trim();
      
      Object.keys(sheetData.podMetadata).forEach(function(podKey) {
        var metadata = sheetData.podMetadata[podKey];
        
        if (metadata.trustId && 
            metadata.trustId.trim() !== "" && 
            metadata.trustId.trim().toUpperCase() !== "#N/A" && 
            metadata.trustId.trim().toUpperCase() !== "NAN") {
          
          var escapedKey = podKey.replace(/[-\/\\^$*+?.()|[\]{}]/g, '\\$&');
          var strictPodRegex = new RegExp("\\b" + escapedKey + "\\b", "i");
          
          if (strictPodRegex.test(cleanTextAreaContent)) {
            var numericId = Number(metadata.trustId.trim());
            
            if (!isNaN(numericId)) {
              if (validStatusCastIds.has(numericId)) {
                if (realComponentIds.indexOf(numericId) === -1) {
                  realComponentIds.push(numericId);
                }
              } else {
                droppedComponentLog.push("'" + podKey + "' (ID: " + numericId + ")");
              }
            }
          }
        }
      });
    }

    if (realComponentIds.length === 0) {
      var errorLogString = droppedComponentLog.length > 0 ? "\nRejected elements discovered: " + droppedComponentLog.join(', ') : "";
      throw new Error("Aborted submission: No active, matching component IDs found on your StatusCast platform instance for these selections." + errorLogString);
    }

    var structuredEmailHtmlMessage = buildUnifiedHtmlBody(formData, true);
    
    var incidentPayload = {
      "title": uniqueTitle,
      "subject": customSubjectLine, 
      "isPublished": false, 
      "treatAsDowntime": false,     
      "sendNotifications": false,        
      "externalId": "",
      "affectedComponents": realComponentIds, 
      "affectedGroups": [],
      "customFields": [],
      "posts": [{
        "text": structuredEmailHtmlMessage,              
        "date": currentIsoDate,       
        "isPublished": false,          
        "postType": "Informational"
      }]
    };

    if (statusType === "Unavailable") {
      incidentPayload.incidentType = "ServiceUnavailable";
      incidentPayload.newComponentStatus = "Unavailable";
      incidentPayload.posts[0].postType = "Investigating";
    } 
    else if (statusType === "Performance Degraded" || statusType === "Performance Degradation") {
      incidentPayload.incidentType = "ServiceUnavailable";
      incidentPayload.newComponentStatus = "DegradedPerformance";
      incidentPayload.posts[0].postType = "Investigating";
    } 
    else if (statusType === "Maintenance") {
      var startWindow = new Date(now.getTime() + (24 * 60 * 60 * 1000)); 
      var endWindow = new Date(now.getTime() + (28 * 60 * 60 * 1000));   

      incidentPayload.incidentType = "ScheduledMaintenance";
      incidentPayload.newComponentStatus = "Maintenance";
      incidentPayload.startDate = startWindow.toISOString();
      incidentPayload.endDate = endWindow.toISOString();
      incidentPayload.assignedUserEmail = currentUserEmail;
      incidentPayload.scheduledMaintenance = {
        "estimatedDurationMinutes": 240,
        "autoClose": false, 
        "reminders": [
          { "amount": 4, "unit": "Hours" },
          { "amount": 2, "unit": "Days" }
        ]
      };
      incidentPayload.posts[0].postType = "Informational";
    } 
    else { 
      incidentPayload.incidentType = "Informational";
      incidentPayload.newComponentStatus = "Informational";
      incidentPayload.posts[0].postType = "Informational";
    }
    
    var incidentOptions = {
      "method": "post",
      "contentType": "application/json",
      "headers": { "Authorization": authorizationValue, "accept": "application/json" },
      "payload": JSON.stringify(incidentPayload),
      "muteHttpExceptions": true
    };
    
    var response = UrlFetchApp.fetch(incidentUrl, incidentOptions);
    if (response.getResponseCode() >= 400) throw new Error("StatusCast Server Rejection: " + response.getContentText());
    
    return { status: 'success' };
  } catch (err) {
    throw new Error(err.message);
  }
}
