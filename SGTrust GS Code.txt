/**
 * DYNAMIC POD DATA INGESTION ENGINE
 */
function fetchPodDataFromSheet() {
try {
    let rows;
    try {
      // 1. Unified the variable name to 'sheetUrl'
      const sheetUrl = "https://docs.google.com/spreadsheets/d/1IHhaFXSFqKtwOfmBANlxss8Hp738SyRPncjhAbUaJo/edit?gid=1532506294#gid=1532506294";
      
      // 2. Used openByUrl since we are passing the full web link
      const ss = SpreadsheetApp.openByUrl(sheetUrl);
      
      // 3. Set a default fallback sheet (index 0)
      let sheet = ss.getSheets()[0];
      
      // 4. Safely filter for the specific tab ID if it exists
      const targetSheets = ss.getSheets().filter(s => s.getSheetId() == 1532506294);
      if (targetSheets.length > 0) {
        sheet = targetSheets[0];
      }

      const lastRow = sheet.getLastRow();
      const lastCol = sheet.getLastColumn();
      
      // 5. Added a safety check just in case the target sheet is empty
      if (lastRow > 0 && lastCol > 0) {
        rows = sheet.getRange(1, 1, lastRow, lastCol).getValues();
      } else {
        rows = [];
      }
      
    } catch (sheetErr) {
      console.error("Sheet baseline disconnect handled: " + sheetErr.toString());
      return {
        status: 'success',
        individualPods: ["VV3-3057", "VV3-3099"],
        podMetadata: {"VV3-3057": {region: "APAC", release: "General", product: "Vault", trustId: ""}},
        uniqueRegions: ["US", "EU", "APAC"],
        uniqueReleases: ["General"],
        uniqueProducts: ["Safety", "LIMS", "RegOne", "Vault"]
      };}
    

    const headers = rows[0].map(h => String(h).trim().toUpperCase());
    
    const podColIndex = headers.indexOf("POD");          
    const regionColIndex = headers.indexOf("REGION");    
    const releaseColIndex = headers.indexOf("RELEASE");  
    const productColIndex = headers.indexOf("PRODUCT");
    const podGroupColIndex = headers.indexOf("VAULT POD GROUP"); 
    const trustIdColIndex = headers.indexOf("TRUST ID"); // Maps Column K
    
    let podValues = [];
    let podMetadata = {};
    let regionsSet = new Set();
    let releasesSet = new Set();
    let productsSet = new Set();

    for (let i = 1; i < rows.length; i++) {
      let currentPod = (podColIndex !== -1) ? String(rows[i][podColIndex]).trim() : "";
      let currentRegion = (regionColIndex !== -1) ? String(rows[i][regionColIndex]).trim() : "";
      let currentRelease = (releaseColIndex !== -1) ? String(rows[i][releaseColIndex]).trim() : "";
      let currentProduct = (productColIndex !== -1) ? String(rows[i][productColIndex]).trim() : "";
      let currentPodGroup = (podGroupColIndex !== -1) ? String(rows[i][podGroupColIndex]).trim() : "";
      let currentTrustId = (trustIdColIndex !== -1) ? String(rows[i][trustIdColIndex]).trim() : "";

      if (currentPod !== "" && currentPodGroup !== "") {
        podValues.push(currentPod);
        
        if (!currentRegion) currentRegion = "Unknown";
        if (!currentRelease) currentRelease = "Unknown";
        if (!currentProduct) currentProduct = "Unknown";

        if (currentRegion !== "NA") regionsSet.add(currentRegion);
        releasesSet.add(currentRelease);
        
        if (currentProduct !== "Unknown" && currentProduct !== "Vault Basics" && currentProduct !== "") {
          productsSet.add(currentProduct);
        }

        podMetadata[currentPod] = {
          region: currentRegion,
          release: currentRelease,
          product: currentProduct,
          trustId: currentTrustId
        };
      }
    }
    
    let finalUniqueProducts = [...productsSet].sort();
    if (finalUniqueProducts.length === 0) {
      finalUniqueProducts = ["Safety", "LIMS", "RegOne", "Vault"];
    }
    
    return {
      status: 'success',
      individualPods: [...new Set(podValues)].sort((a,b) => String(a).localeCompare(String(b), undefined, {numeric: true})),
      podMetadata: podMetadata,
      uniqueRegions: [...regionsSet].sort(),
      uniqueReleases: [...releasesSet].sort(),
      uniqueProducts: finalUniqueProducts,
      footerHtml: getUnifiedFooterText() // Safe handshake to the frontend UI
    };
  } catch (err) {
    return { status: 'error', message: err.toString() };
  }
}

function getSenderIdByEmail(apiKey, targetEmail) {
  const options = {
    "method": "get",
    "headers": { "Authorization": "Bearer " + apiKey },
    "muteHttpExceptions": true
  };
  
  const response = UrlFetchApp.fetch("https://api.sendgrid.com/v3/verified_senders", options);
  if (response.getResponseCode() !== 200) {
    throw new Error("Unable to read Sender Identity records from SendGrid: " + response.getContentText());
  }
  
  const data = JSON.parse(response.getContentText());
  const sendersList = data.results || data || [];
  const match = sendersList.find(s => s.from_email && s.from_email.toLowerCase() === targetEmail.toLowerCase());
  if (!match) throw new Error("Could not find a SendGrid verified identity matching: " + targetEmail);
  return match.id;
}

function processPushRequest(formData) {
  try {
    if (!formData) formData = {};
    const scriptProperties = PropertiesService.getScriptProperties();
    const SENDGRID_API_KEY = scriptProperties.getProperty('SendGridAPIKeyID');
    if (!SENDGRID_API_KEY) throw new Error("SendGrid API Key parameter context not discovered.");

    const rawVersion = formData.version || "26R1.0.22";
    const cleanSubject = formData.subjectLine || "Veeva Vault Maintenance Release Notification";
    const cleanFromSender = formData.fromSender || "no-reply@vault.veeva.com"; 

    const targetSenderId = getSenderIdByEmail(SENDGRID_API_KEY, cleanFromSender);
    const compiledSubjectWithVersion = cleanSubject.trim() + " - " + rawVersion.trim();
    const finalHtmlBody = buildUnifiedHtmlBody(formData, false);

    const payload = {
      "name": compiledSubjectWithVersion, 
      "email_config": { 
        "subject": compiledSubjectWithVersion, 
        "html_content": finalHtmlBody, 
        "generate_plain_content": true,
        "custom_unsubscribe_url": "https://platform.veevavault.help/en/lr/7239/#notification-email",
        "sender_id": targetSenderId,
        "tracking_settings": {
          "open_tracking": { "enable": false },
          "click_tracking": { "enable": false }
        }
      }
    };

    const options = {
      "method": "post",
      "headers": { "Authorization": "Bearer " + SENDGRID_API_KEY, "Content-Type": "application/json" },
      "payload": JSON.stringify(payload),
      "muteHttpExceptions": true
    };

    const response = UrlFetchApp.fetch("https://api.sendgrid.com/v3/marketing/singlesends", options);
    if (response.getResponseCode() >= 400) throw new Error("SendGrid Refused Payload: " + response.getContentText());
    return { status: 'success' };
  } catch (err) {
    throw new Error(err.message);
  }
}
